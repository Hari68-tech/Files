-----------------------------------------------------------------------------------------------------------------------
Syntax of declaring an Array
-----------------------------------------------------------------------------------------------------------------------

		datatype[] array_var = {val1,val2,...valn};	


-------------------------------------------------------------------------------------------------------------------------	
Accessing elements present in a Array 
------------------------------------------------------------------------------------------------------------------------

		array_var[index];

---------------------------------------------------------------------------------------------------------------------
Replacing existing elements in a Array
------------------------------------------------------------------------------------------------------------------------

		array_var[index] = new_value;

----------------------------------------------------------------------------------------------------------------------
Counting no. of elements present in a Array 
----------------------------------------------------------------------------------------------------------------------

		array_var.length;

----------------------------------------------------------------------------------------------------------------------
Array Example
----------------------------------------------------------------------------------------------------------------------		
		String[] cars = {"car1","car2","car3"};
		
		// accessing elements present in a Array
		
		System.out.println(cars[0]);
		System.out.println(cars[1]);
		System.out.println(cars[2]);
		
		// replacing existing elements
		
		cars[0] ="new car";
		System.out.println("after replacing...");
		System.out.println(cars[0]);
		
		// counting no. of elements
		
		int count =  cars.length;	
		System.out.println(count);
